##2.创建服务器
###http.createServer
```javascript
//引入http内置模块
var http = require("http");
http.createServer(function(request,response){
    //设置响应头;
    response.setHeader("Access-Control-Allow-Origin","*");
    //返回结果
    response.end("Hello World");
}).listen(12345)
//端口号有范围限制0~65535
```
注意点：<br>
1.是否跨域<br>
2.每次修改完代码，记得重新用node执行后端.js文件<br>
3.创建的服务器是指向xxx.js执行后的域名和端口号<br>
odejs跑一个http.js(代替了php+apache)

post请求:接收传递过来的参数;
```Javascript
//接收传递过来的参数;
    var post ="";
    request.on("data",function(msg){//data是固定的;
        post += msg;
    });

    request.on('end',function(){
        post = querystring.parse(post); =>引入querystring模块处理参数;
        console.log(post.type);  //=>返回值:post;前端传递data:{type:post}
    });
```

###2)POST/GET请求 与 处理参数
```Javascript
//使用bodyParser模块;处理post请求传递过来的参数;
var bodyParser = require('body-parser');
app.use(bodyParser.urlencoded({extended:false}));

//拓展:处理JSON请求app.use(bodyParser.json());

// ------------get请求-------------------
app.get('/',function(req,res){
    console.log(req.query); //获取get请求发送过来的参数;
    //设置响应头;
    res.append("Access-Control-Allow-Origin","*");

    res.send('get')
    
});
// ---------post请求---------------------
app.post('/',function(req,res){
    console.log(req.body);  //获取Post请求发送过来的参数;
    //设置响应头;
    res.append("Access-Control-Allow-Origin","*");

    res.send('post')
    
});
// ----------All请求----------------------
app.all('/',function(req,res){
    //设置响应头;
    res.append("Access-Control-Allow-Origin","*");

    res.send('all');
})
```

##5.Node.js服务器代理(http.request)
```Javascript
//node代理PHP服务器;
 var http = require("http");

//理解:相当于nodejs向php发送了"ajax请求",用http.request方法;
//http://localhost/three/test.php =>PHP的服务器地址;
http.request({
    hostname: 'localhost',
    port: '80',
    path: '/three/test.php?name=heightzhang',
    method: 'GET'
}, function(res) {
    res.setEncoding('utf8');
    var data = "";
    res.on('data', function(chunk){
        data += chunk
    });
    res.on('end', function(){
        //连接服务器后返回的数据
        console.log(data);
    });
}).on('error', function(e) {
    console.log('problem with request: ' + e.message);
}).end();
```
